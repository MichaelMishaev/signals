📦 Implementation Guide

Import Component:

import { ActionButton } from '@/components/shared/sharedbuttons';
// or import specific variants
import {
  UrgentCountdownButton,
  LivePulseButton,
  ProfitAlertButton,
  RocketLaunchButton
} from '@/components/shared/sharedbuttons';

Basic Usage:

// Using main component with variant
<ActionButton
  variant="urgent-countdown"
  onClick={handleOpenTrade}
/>

// Using specific variant component
<UrgentCountdownButton
  onClick={handleOpenTrade}
  fullWidth
/>

Available Props:

interface ActionButtonProps {
  variant?: 'urgent-countdown' | 'live-pulse' | 'profit-alert' | 'rocket-launch';
  onClick?: () => void;
  disabled?: boolean;
  className?: string;
  fullWidth?: boolean;
  size?: 'sm' | 'md' | 'lg';
  customText?: string;
  children?: React.ReactNode;
}

Examples:

// Custom text
<ActionButton variant="live-pulse" customText="START NOW" />

// Different sizes
<ActionButton variant="profit-alert" size="lg" />

// Full width
<ActionButton variant="rocket-launch" fullWidth />

// With custom handler
<ActionButton
  variant="urgent-countdown"
  onClick={() => {
    analytics.track('trade_opened');
    openTradeModal();
  }}
/>
